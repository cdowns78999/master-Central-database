Swipe Deck

A · tinder-style

Drag right to accept, left to pass. Cards stack, peel away, next rises.

card 1 of 3
view code 
(()=>{
  const cards=[/*3 cards*/];
  const deck=document.getElementById('deckA');
  let i=0; render();
  function render(){
    deck.innerHTML='';
    cards.slice(i,i+3).forEach((c,k)=>{
      const el=make(c); el.dataset.state=k; deck.appendChild(el);
    });
  }
  function advance(dir){
    const top=deck.firstElementChild; if(!top) return;
    top.style.setProperty('--dx', dir*320+'px');
    top.style.setProperty('--dr', dir*18+'deg');
    top.classList.add('gone');
    setTimeout(()=>{ i=Math.min(i+1,cards.length); render(); }, 320);
  }
})();
