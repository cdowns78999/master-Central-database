];

let soIdx = 0;
let soIsOpen = false;

function soRender(idx) {
  const s = SO_STEPS[idx];
  const els = ['soCounter','soNum','soTitle','soDesc','soBody'];
  const vals = [
    'STEP ' + (idx+1) + ' OF ' + SO_STEPS.length,
    s.num, s.title, s.desc, s.body
  ];
  els.forEach((id,i) => {
    const el = document.getElementById(id);
    if (el) el.textContent = vals[i];
  });
  document.querySelectorAll('.so-dot').forEach((d,i) => {
    d.classList.toggle('active', i === idx);
  });
}

function soOpenOverlay() {
  soIdx = 0;
  soRender(0);
  const ov = document.getElementById('step-overlay');
  if (ov) { ov.classList.add('active'); soIsOpen = true; }
}

function soCloseOverlay() {
  const ov = document.getElementById('step-overlay');
  if (ov) { ov.classList.remove('active'); soIsOpen = false; }
}

window.addEventListener('DOMContentLoaded', () => {
  const prev  = document.getElementById('soPrev');
  const next  = document.getElementById('soNext');
  const close = document.getElementById('soClose');
  const dots  = document.querySelectorAll('.so-dot');

  if (prev) prev.addEventListener('click', () => {
    soIdx = (soIdx - 1 + SO_STEPS.length) % SO_STEPS.length;
    soRender(soIdx);
  });
  if (next) next.addEventListener('click', () => {
    soIdx = (soIdx + 1) % SO_STEPS.length;
    soRender(soIdx);
  });
  if (close) close.addEventListener('click', soCloseOverlay);

  dots.forEach((d, i) => d.addEventListener('click', () => { soIdx = i; soRender(i); }));

  document.addEventListener('keydown', e => {
    if (!soIsOpen) return;
    if (e.key === 'ArrowRight') { soIdx = (soIdx+1) % SO_STEPS.length; soRender(soIdx); }
    if (e.key === 'ArrowLeft')  { soIdx = (soIdx-1+SO_STEPS.length) % SO_STEPS.length; soRender(soIdx); }
    if (e.key === 'Escape') soCloseOverlay();
  });
});
// ══ End Step Overlay System ══════════════════════════
